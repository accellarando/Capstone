static void setup_image_window() {
    if (image_window == NULL) {
        image_window = gtk_window_new(GTK_WINDOW_TOPLEVEL);
        gtk_window_set_title(GTK_WINDOW(image_window), "Image Display");
        gtk_window_set_default_size(GTK_WINDOW(image_window), GRID_SIZE, GRID_SIZE);
        g_signal_connect(G_OBJECT(image_window), "destroy", G_CALLBACK(gtk_widget_destroy), NULL);

        GtkWidget *image_display = gtk_image_new();
        gtk_container_add(GTK_CONTAINER(image_window), image_display);
        gtk_widget_show_all(image_window);
    }
}
