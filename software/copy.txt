void setup_selector() {
    // Create drawing area
    selector_area = gtk_drawing_area_new();
    gtk_widget_set_size_request(selector_area, WINDOW_WIDTH / 2, WINDOW_HEIGHT);
    gtk_box_pack_start(GTK_BOX(vbox), selector_area, TRUE, TRUE, 0);

    // Create image display area
    image_display_area = gtk_image_new();
    gtk_widget_set_size_request(image_display_area, GRID_SIZE, GRID_SIZE);

    // Create image window
    image_window = gtk_window_new(GTK_WINDOW_TOPLEVEL);
    gtk_window_set_title(GTK_WINDOW(image_window), "Image Display");
    gtk_window_set_default_size(GTK_WINDOW(image_window), GRID_SIZE, GRID_SIZE);
    g_signal_connect(G_OBJECT(image_window), "destroy", G_CALLBACK(gtk_widget_destroy), NULL);
    gtk_container_add(GTK_CONTAINER(image_window), image_display_area);

    // Hide both selector_area and image_window initially
    gtk_widget_hide(selector_area);
    gtk_widget_hide(image_window);
}
